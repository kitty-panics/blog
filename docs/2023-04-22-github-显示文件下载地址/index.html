Github 显示直接文件下载地址

Kitty • Panics  included in JavaScript
 2023-04-22  294 words   One minute 
Contents

Github 网页下载文件的体验太糟糕了，得点击好几次，有时我只是想复制下载链接而已。

页面中文件列表最右边的目录、文件图标也没有什么用，直接将其替换成下载按钮。

// ==UserScript==
// @name         Github: 文件显示直接下载按钮
// @version      2023.04.22
// @description  文件显示直接下载按钮。
// @author       kitty-panics
// @match        *://github.com/*
// @icon         https://github.com/favicon.ico
// @grant        window.onurlchange
// ==/UserScript==

(function() {
    'use strict';

    function addDownloadButton(elem, path) {
        const url = 'https://github.com' + path.replace('blob', 'raw');
        const button = document.createElement('a');
        button.href = url;
        button.target = '_blank';
        button.innerText = '🡇';
        elem.parentNode.replaceChild(button, elem);
    }

    function parsingSVG() {
        const allSVG = document.querySelectorAll('svg[aria-label="File"]');
        if (allSVG.length === 0) {
            const allSVG = document.querySelectorAll('svg[class="color-fg-muted"]');
            for (let i = 0; i <= allSVG.length; i++) {
                if (i % 2 !== 0) {
                    const filePath = allSVG[i].nextElementSibling.querySelector('h3 div a').getAttribute('href');
                    addDownloadButton(allSVG[i], filePath);
                }
            }
        } else {
            allSVG.forEach(function(svg) {
                const filePath = svg.parentElement.nextElementSibling.querySelector('span a').getAttribute('href');
                addDownloadButton(svg, filePath);
            });
        }
    }

    window.onload = function() {
        setTimeout(() => parsingSVG(), 1500);

        if (window.onurlchange === null) {
            window.addEventListener('urlchange', function() {
                setTimeout(() => parsingSVG(), 1500);
            });
        }
    };

})();
