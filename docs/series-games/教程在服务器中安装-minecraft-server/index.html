# 向所有在线 OP 发送通过 RCON 执行的命令的输出
broadcast-rcon-to-ops=true
# 服务端发送给客户端的世界数据量 (玩家各个方向 (半径) 上的区块数量，是服务端的可视距离)
view-distance=16
# 启用将使 MBean 使用以毫秒为单位时间计时
enable-jmx-monitoring=false
# 将服务器与一个特定 IP 绑定
server-ip=127.0.0.1
# [可选] 在使用 require-resource-pack 时，在资源包提示界面显示自定义信息
resource-pack-prompt=
# 设置 RCON 远程访问的端口号
rcon.port=25575
# 定义默认游戏模式, [survival, creative, adventure, spectator]
gamemode=survival
# 设置服务器监听的 IPv4 端口号 (如果服务器在使用 NAT 的网络中运行，该端口必须被转发)
server-port=25565
# 允许玩家进入下界
allow-nether=true
# 是否启用命令方块
enable-command-block=false
# 是否允许远程访问服务器控制台 (RCON 协议传输数据时没有加密)
enable-rcon=false
# 启用后区块文件以同步模式写入
sync-chunk-writes=true
# 允许使用 GameSpy4 协议的服务器监听器，用于获取服务器信息
enable-query=false
# 设定使用 /op 命令时 OP 的权限等级 (所有存档会从之前的存档继承能力和命令)
op-permission-level=4
# 如果服务器发送的 ISP/AS 和 Mojang 的验证服务器的不一样，玩家会被踢出
prevent-proxy-connections=false
# [可选] 一个资源包的 URI，玩家可选择是否使用该资源包
resource-pack=
# 实体需要距离玩家有多近才会将数据包发送给客户端
entity-broadcast-range-percentage=128
# 此值将作为世界名称及其文件夹名
level-name=world
# 设置 RCON 远程访问的密码
rcon.password=
# 如果不为 0，服务器将在玩家的空闲时间达到设置的时间 (分钟) 时将玩家踢出服务器
player-idle-timeout=15
# 玩家客户端的多人游戏服务器列表中显示的服务器信息 (显示于名称下方)
motd=A Minecraft Server
# 设置监听服务器的端口号
query.port=25565
# 强制玩家加入时为默认游戏模式
force-gamemode=true
# 玩家被踢出服务器前，可以发送的数据包数量 (为 0 表示关闭此功能)
rate-limit=0
# 如果设为 true，会忽略服务器的难度并设为 hard，玩家在死后会自动切换至旁观模式
hardcore=false
# 启用服务器的白名单
white-list=true
# 向所有在线 OP 发送所执行命令的输出
broadcast-console-to-ops=true
# 是否允许 PvP (只有在允许 PvP 时玩家自己的箭才会受到伤害)
pvp=true
# 决定是否生成村民
spawn-npcs=true
# 决定动物是否可以生成
spawn-animals=true
# 是否允许服务端定期发送统计数据到 http://snoop.minecraft.net
snooper-enabled=false
# 定义服务器的游戏难度, [peaceful, easy, normal, hard]
difficulty=normal
# 设定函数的默认权限等级
function-permission-level=2
# 允许 n-1 字节的数据包正常发送 (如果数据包为 n 字节或更大时会进行压缩)
network-compression-threshold=256
# 
text-filtering-config=
# 启用时玩家会被提示是否启用服务器资源包 (如果拒绝则会被服务器断开连接)
require-resource-pack=false
# 决定攻击型生物 (怪物) 是否可以生成
spawn-monsters=true
# 设置每个 tick 花费的最大毫秒数
max-tick-time=60000
# 在服务器上强制执行白名单 (不在白名单的用户将在服务器重新加载白名单文件后从服务器踢出)
enforce-whitelist=true
# 是否使用针对 Linux 平台的数据包收发优化
use-native-transport=true
# 服务器同时能容纳的最大玩家数量
max-players=5
# 资源包的 SHA-1 值 (必须为小写十六进制，建议填写它)
resource-pack-sha1=
# 通过将该值进行 2x+1 的运算来决定出生点的保护半径
spawn-protection=16
# 是否启用正版验证
online-mode=false
# 使服务器在服务器列表中看起来是 "在线" 的
enable-status=true
# 允许玩家在安装添加飞行功能的 MOD 前提下在生存模式下飞行
allow-flight=false
# 可设置世界边界的最大半径值 (单位为方块)
max-world-size=29999984
