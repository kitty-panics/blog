笔顺数据

笔顺数据采用 CNMan/UnicodeCJK-WuBi ，这是网友做的整合版，可能是全网最全的。 Unicode CJK 收录了近 10 万多个汉字，而此数据库就有近 9 万个汉字的笔顺数据。

所有汉字

先看结果，基本上算是一样多，奇数只比偶数多一丢丢。

# -*- coding: utf-8 -*-
#
# Prepare:
# cat CJK*.txt > stroke-order.txt

import csv

odd, even = 0, 0

with open('stroke-order.txt', 'r') as file:
    csv_reader = csv.reader(file, delimiter=',')
    for row in csv_reader:
        column = row[5]
        if column.strip():
            if len(column) % 2 == 0:
                odd += 1
            else:
                even += 1

print(f'奇数数量为：{odd}')
print(f'偶数数量为：{even}')

# -*- coding: utf-8 -*-

import os, csv

def Counter(csv_reader, cjk):
    sc_odd, sc_even, tc_odd, tc_even = 0, 0, 0, 0
    for row in csv_reader:
        sc_column, tc_column = row[5], row[6]
        if sc_column.strip():
            if len(sc_column) % 2 == 0:
                sc_odd += 1
            else:
                sc_even += 1
        if tc_column.strip():
            if len(tc_column) % 2 == 0:
                tc_odd += 1
            else:
                tc_even += 1
    print(f'| {cjk} | {sc_odd} | {sc_even} | {tc_odd} | {tc_even} |')
