为深切哀悼在抗击新冠肺炎斗争中的牺牲烈士和逝世同胞，我的个人网站将全站变灰 3 天。

愿逝者安息，愿生者奋发，愿祖国昌盛。

网页变灰的 UserScript

为此写了一个在公祭日时将所有网页变灰的用户脚本。

PC 端的浏览器

支持所有 Firefox 系和 Chromium 系浏览器，前提是已安装 Tampermonkey 或之类的浏览器扩展。

复制下面的代码，并手动添加到 Tampermonkey 中。

// ==UserScript==
// @name         国家公祭日
// @namespace    https://tampermonkey.net/
// @version      1.1
// @description  在国家公祭日时将所有网页变灰
// @author       小牛吃羊排
// @match        *://*/*
// @grant        none
// @run-at       document-idle
// ==/UserScript==

(function() {
    'use strict';

    // 加载样式
    function load_Style() {
        let new_style = document.createElement('style');
        new_style.type = 'text/css';
        new_style.appendChild(document.createTextNode('body{ filter: grayscale(100%); }'));
        let get_head = document.getElementsByTagName('head')[0];
        get_head.appendChild(new_style);
    }

    // 获取月与日
    let month = new Date().getMonth();
    let day = new Date().getDate();
    let month_day = String(month) + String(day);

    // 判断日期
    if(month_day == "34") { // 04 月 04 日，清明节，中国传统节日，也是最重要的祭祀节日之一
        load_Style();
    } else if(month_day == "412") { // 05 月 12 日，四川汶川发生特大地震
        load_Style();
    } else if(month_day == "818") { // 09 月 18 日，九·一八事变，是日本帝国主义侵华的开端
        load_Style();
    } else if(month_day == "1113") { // 12 月 13 日，南京大屠杀死难者国家公祭日
        load_Style();
    } else {
        console.log('[Parameter Error] month_day: ' + month_day);
    }

})();
