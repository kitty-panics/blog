被速率限制时自动切换 Github 镜像站

Kitty • Panics  included in JavaScript
 2022-01-12  885 words   2 minutes 
Contents
  • 使用
  • 代码

由于众所周知的原因，访问 Github 变得越来越困难，幸好还有大量的镜像站可用，但镜像站使用的人太多， 很容易被速率限制。

被速率限制时，只能等待一会再访问 (等待总是煎熬的)，要不换到另一个镜像站 (输入一大串 URL 又太麻烦)。

秉持着任何 会手动重复三次的操作都应自动化 的原则，于是就有了这个 Tampermonkey 脚本。

使用

  1. 安装 Redirector 和 Tampermonkey 浏览器扩展。
  2. 在 Redirector 中配置将 Github 重定向至随便一个镜像站。
    • 这一步不用 Tampermonkey 脚本实现的原因是因为只有网站能正常开始访问到时 Tampermonkey 脚本才开始加载，但 Github 并不总是能正常访问。
  3. 在 Tampermonkey 中添加下面的脚本。

代码

特别需要注意的是，为了自己的 Github 帐号安全，不要在镜像站中登录帐号。

// ==UserScript==
// @name         # Github - 被速率限制时自动切换镜像站 #
// @namespace    https://tampermonkey.net
// @version      2.6
// @description  被速率限制时自动切换 Github 镜像站
// @author       kitty-panics
// @match        https://xx.yy.com/*
// @match        https://xxyy.zz.net/*
// @match        https://xx.xx.cc/*
// @match        https://xx.yy-zz.xyz/*
// @grant        window.close
// @run-at       document-idle
// ==/UserScript==

(function() {
    'use strict';

    // 将所有 <a> 标签设置为在新标签页打开
    // 防止因 rate-limit 导致长时间无响应
    function open_In_New_Tab() {
        let a_tag = document.getElementsByTagName('a');
        for (let i = 0; i < a_tag.length ; i++) {
            a_tag[i].addEventListener('click', function () {
                // 方案一
                window.open(this.href, '_blank');
                window.close(); // 即使在新标签页打开，原标签页也会响应，所以直接 close
                // 方案二
                //window.open(this.href, '_self');
            }, false);
        }
    }
    open_In_New_Tab();

    // 返回 URL 中的 Host Name 和 PATH
    // indexOf 从 8 开始检索是因为 'https://' 刚好 8 个字符
    function get_HN_Path(url, hn_path) {
        if (hn_path == 'hostname') {
            return url.substring(0, url.indexOf('/', 8));
        } else if (hn_path == 'path') {
            return url.substring(url.indexOf('/', 8));
        } else {
            console.log('Unknown "hn_path": ' + hn_path);
        }
    }

    // 通过 Value 查 Index
    function get_Index_By_Value(array, value) {
        return array.map(function (item) {
            return item;
        }).indexOf(value);
    }

    // 每次轮询等待 1 秒
    function delay_Time(hostname, path) {
        setTimeout(function () {
            window.location.href = hostname + path;
        }, 1000)
    }


    // 可用的镜像站 (请自行添加)
    const mirror_site = ['https://xx.yy.com',
                         'https://xxyy.zz.net',
                         'https://xx.xx.cc',
                         'https://xx.yy-zz.xyz']


    // 获取当前页面标题、URL
    const site_title = document.title;
    const site_url = window.location.href;

    // 提取 URL 中的 Host Name 和 PATH
    const url_hostname = get_HN_Path(site_url, 'hostname');
    const url_path = get_HN_Path(site_url, 'path');

    // 先判断是否是镜像站，如果 "不是" 就重定向至镜像站
    // 由于 Tampermonkey 机制限制，并不能很好实现这一功能，转由 Redirector 扩展实现
    // Redirector 主页：https://einaregilsson.com/redirector
    //
    // 再判断镜像站是否被 rate-limit，如果 "是" 就轮询多个镜像站
    if (site_title == 'Rate limit · GitHub') {
        let index = get_Index_By_Value(mirror_site, url_hostname);
        if (index != -1) {
            let mirror_len = mirror_site.length;
            mirror_len--;
            if (index < mirror_len) {
                index++;
                delay_Time(mirror_site[index], url_path);
            } else {
                delay_Time(mirror_site[0], url_path);
            }
        } else {
            console.log('Unknown "url_hostname": ' + url_hostname);
        }
    }

})();
