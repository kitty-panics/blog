Github 显示绝对时间

Kitty • Panics  included in JavaScript
 2023-03-25  464 words   One minute 
Contents
  • 脚本

很奇怪 Github 的产品设计逻辑，用户都是开发者，很多东西需要精确表达，在时间设计上偏偏用相对时间， 有时真的很碍事，尽管可以通过悬停鼠标来显示绝对时间。

只能写个 Tampermonkey 脚本将其改成绝对时间。

脚本

当然这个脚本并不是很完美，当然这也不是我的问题。

因为 Github 会自动高频率的刷新时间，比如才提交的文件显示的时间是 now，那么会每秒刷新一次，直到 1 min 后变成 1 minutes ago，然后此文件的刷新时间会每分钟刷新一次。

但时间相关的方法不能被 Hook 掉，会有其它功能依赖这些方法，尽管不能从功能逻辑上看出两者能有啥依赖关系。

// ==UserScript==
// @name         # Github: Replace relative-time with real-time #
// @version      2023.03.25
// @description  SO, RELATIVE-TIME, F**K YOU!
// @author       kitty-panics
// @match        *://*.github.com/*
// @icon         https://github.com/favicon.ico
// ==/UserScript==

(function() {
    'use strict';

    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone;
    const locale = navigator.language;

    function fmtDate(dateTime, tz, locale, inclSec) {
        const options = { timeZone: tz, hour12: false, year: 'numeric', month: '2-digit', day: '2-digit', hour: '2-digit', minute: '2-digit' };
        if (inclSec) options.second = '2-digit';
        // 'az':   Use ISO 8601 format (YYYY-MM-DD HH:mm:ss).
        // locale: Use locale format.
        return new Date(dateTime).toLocaleString('az', options);
    }

    function replaceTime(inclSec = true) {
        document.querySelectorAll('relative-time').forEach(function(relTime) {
            const getDateTime = relTime.getAttribute('datetime');
            relTime.shadowRoot.innerHTML = fmtDate(getDateTime, tz, locale, inclSec);
            const parRelTime = relTime.parentNode;
            if (parRelTime && parRelTime.style.width) {
                parRelTime.style.width = `${relTime.offsetWidth}px`;
            }
        });
    }

    function replaceTimeSearch() {
        document.querySelectorAll('span[title]').forEach(span => {
            span.innerText = fmtDate(span.getAttribute('title'), tz, locale);
            span.parentNode.style.maxWidth = `${span.offsetWidth + 1}px`;
        });
    }

    window.onload = function() {
        const isSearch = window.location.href.includes('https://github.com/search?');
        // Execute "replaceTime()" twice, with an interval of 2s.
        // If your network is slow, increase this value.
        isSearch ? replaceTimeSearch() : (replaceTime(), setTimeout(() => replaceTime(), 2000));
    };

})();
